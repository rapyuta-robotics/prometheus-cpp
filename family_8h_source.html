151  auto metrics_iter = metrics_.find(hash);
152 
153  if (metrics_iter != metrics_.end()) {
154 #ifndef NDEBUG

156  assert(labels_iter != labels_.end());
157  const auto& old_labels = labels_iter->second;

161  } else {
162 #ifndef NDEBUG

164  auto& label_name = label_pair.first;
